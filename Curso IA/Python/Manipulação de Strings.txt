upper(): variavel a = 'casaco'; a.upper() - vai deixar a string em maiuscula;

lower(): variavel a = 'casaco'; a.lower() - vai deixar a string em minuscula;

capitalize(): variavel a = 'casaco'; a.capitalize() - vai deixar a inicial maiuscula(Casaco);

metade_palavra = a[0:3] - vai pegar a palavra casaco e vai até o indice 2 (cas);

metade_palavra = a[3:] - vai pegar a palavra casaco e vai até o final dela (aco);

b = a.replace('aco', 'inha') - vai trocar as letras aco da palavra "casaco" por "inha",
formando "casinha"

c = a.replace('o', 'a') - vai trocar onde tem "o" e colocar "a", fazendo "casaco" virar 
"casaca";

c.find('s') - vai retornar o indice "2" pois é a posição onde se encontra a letra "s";

c.find('a') - vai retornar o indice "1" pois é a posição onde se encontra a primeira 
letra "a";

c.find('b') - vai retornar -1 pois a letra "b" não existe na palavra "casaca";

f = e.strip() - com essa função sér possível remover os espaços em branco antes e depois 
da string;
